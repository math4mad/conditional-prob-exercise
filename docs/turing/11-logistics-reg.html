3.1 define model

Code 
@model function logistic_regression(x, labels)
    m ~ Normal(0, 1)
    b ~ Normal(0, 1)
    sigma ~ Gamma(1, 1)

    n = length(x)
    y = tzeros(n)
    sigmoid = tzeros(n)
    for i in 1:n
        y[i] ~ Normal(m * x[i] + b, sigma)
        sigmoid[i] = logistic(y[i])
        labels[i] ~ Bernoulli(sigmoid[i])
    end

    # Post predictive of labels
    labels_post = tzeros(n)
    for i in 1:n
        labels_post[i] ~ Bernoulli(sigmoid[i])
    end
end
